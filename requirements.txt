# ─────────────────────────────────────────────
def calc_rsi(series, period=14):
    delta = series.diff()
    gain = delta.clip(lower=0).rolling(period).mean()
    loss = (-delta.clip(upper=0)).rolling(period).mean()
    rs = gain / loss.replace(0, np.nan)
    return 100 - (100 / (1 + rs))


def calc_macd(series, fast=12, slow=26, signal=9):
    ema_fast = series.ewm(span=fast, adjust=False).mean()
    ema_slow = series.ewm(span=slow, adjust=False).mean()
    macd_line = ema_fast - ema_slow
    signal_line = macd_line.ewm(span=signal, adjust=False).mean()
    histogram = macd_line - signal_line
    return macd_line, signal_line, histogram


def calc_bollinger(series, period=20, std_dev=2):
    sma = series.rolling(period).mean()
    std = series.rolling(period).std()
    upper = sma + std_dev * std
    lower = sma - std_dev * std
    return upper, sma, lower


def calc_indicators(df):
    close = df["Close"].squeeze()
    volume = df["Volume"].squeeze()

    df = df.copy()
    df["RSI"] = calc_rsi(close)
    df["MACD"], df["MACD_Signal"], df["MACD_Hist"] = calc_macd(close)
    df["BB_Upper"], df["BB_Mid"], df["BB_Lower"] = calc_bollinger(close)
    df["MA20"]  = close.rolling(20).mean()
    df["MA50"]  = close.rolling(50).mean()
    df["MA200"] = close.rolling(200).mean()
    df["Vol_MA20"] = volume.rolling(20).mean()
    df["Vol_Spike"] = volume > (df["Vol_MA20"] * 1.5)
    return df

# SIGNAL GENERATION
# ─────────────────────────────────────────────
def generate_signals(df, ticker, name):
    """
    Analyse indicators and return a structured signal dict with
    buy/sell/hold recommendation and reasoning.
    """
    row   = df.iloc[-1]
    prev  = df.iloc[-2]
    close = float(row["Close"])

    signals   = []
    cautions  = []
    score     = 0   # positive = bullish, negative = bearish

    rsi = float(row["RSI"]) if not pd.isna(row["RSI"]) else 50

    # ── RSI ──
    if rsi < 30:
        signals.append(f"RSI {rsi:.1f} — oversold (potential reversal buy)")
        score += 2
    elif rsi < 45:
        signals.append(f"RSI {rsi:.1f} — approaching oversold territory")
        score += 1
    elif rsi > 70:
        cautions.append(f"RSI {rsi:.1f} — overbought (consider taking profit)")
        score -= 2
    elif rsi > 55:
        signals.append(f"RSI {rsi:.1f} — bullish momentum building")
        score += 1

    # ── MACD ──
    if not pd.isna(row["MACD"]) and not pd.isna(row["MACD_Signal"]):
        macd_cross_up   = (float(row["MACD"]) > float(row["MACD_Signal"])) and \
                          (float(prev["MACD"]) <= float(prev["MACD_Signal"]))
        macd_cross_down = (float(row["MACD"]) < float(row["MACD_Signal"])) and \
                          (float(prev["MACD"]) >= float(prev["MACD_Signal"]))
        if macd_cross_up:
            signals.append("MACD bullish crossover — momentum turning positive")
            score += 3
        elif macd_cross_down:
            cautions.append("MACD bearish crossover — momentum turning negative")
            score -= 3
        elif float(row["MACD"]) > float(row["MACD_Signal"]):
            signals.append("MACD above signal line — uptrend confirmed")
            score += 1
        else:
            cautions.append("MACD below signal line — downtrend in place")
            score -= 1

    # ── Bollinger Bands ──
    if not pd.isna(row["BB_Lower"]) and not pd.isna(row["BB_Upper"]):
        bb_lower = float(row["BB_Lower"])
        bb_upper = float(row["BB_Upper"])
        bb_mid   = float(row["BB_Mid"])
        pct_b    = (close - bb_lower) / (bb_upper - bb_lower) if (bb_upper - bb_lower) > 0 else 0.5

        if close < bb_lower:
            signals.append(f"Price below lower Bollinger Band — strong oversold signal, bounce likely")
            score += 2
        elif close > bb_upper:
            cautions.append(f"Price above upper Bollinger Band — overbought, pullback risk")
            score -= 2
        elif pct_b < 0.2:
            signals.append(f"Price near lower Bollinger Band ({pct_b*100:.0f}%B) — support level")
            score += 1

    # ── Moving Averages ──
    ma20  = float(row["MA20"])  if not pd.isna(row["MA20"])  else None
    ma50  = float(row["MA50"])  if not pd.isna(row["MA50"])  else None
    ma200 = float(row["MA200"]) if not pd.isna(row["MA200"]) else None

    if ma20 and ma50:
        if close > ma20 > ma50:
            signals.append("Price above MA20 and MA50 — short and medium trend bullish")
            score += 2
        elif close < ma20 < ma50:
            cautions.append("Price below MA20 and MA50 — short and medium trend bearish")
            score -= 2
        # Golden / Death cross
        if not pd.isna(prev["MA50"]):
            if float(row["MA20"]) > float(row["MA50"]) and float(prev["MA20"]) <= float(prev["MA50"]):
                signals.append("GOLDEN CROSS — MA20 crossed above MA50, major bullish signal")
                score += 4
            elif float(row["MA20"]) < float(row["MA50"]) and float(prev["MA20"]) >= float(prev["MA50"]):
                cautions.append("DEATH CROSS — MA20 crossed below MA50, major bearish signal")
                score -= 4

    if ma200:
        if close > ma200:
            signals.append(f"Price above MA200 (${ma200:.2f}) — long-term uptrend intact")
            score += 1
        else:
            cautions.append(f"Price below MA200 (${ma200:.2f}) — long-term downtrend in place")
            score -= 1

    # ── Volume Spike ──
    if bool(row["Vol_Spike"]):
        vol_ratio = float(row["Volume"]) / float(row["Vol_MA20"])
        if score >= 0:
            signals.append(f"Volume spike ({vol_ratio:.1f}x average) — strong buying interest")
            score += 1
        else:
            cautions.append(f"Volume spike ({vol_ratio:.1f}x average) — high selling pressure")
            score -= 1

    # ── Big dip check (potential reversal) ──
    week_ago  = df["Close"].iloc[-6] if len(df) >= 6 else close
    month_ago = df["Close"].iloc[-22] if len(df) >= 22 else close
    week_chg  = (close - float(week_ago)) / float(week_ago) * 100
    month_chg = (close - float(month_ago)) / float(month_ago) * 100

    if week_chg < -10 and rsi < 40:
        signals.append(f"Big dip: -{abs(week_chg):.1f}% this week with oversold RSI — reversal watch")
        score += 2
    if month_chg < -20 and rsi < 35:
        signals.append(f"Major dip: -{abs(month_chg):.1f}% this month — deep value reversal candidate")
        score += 2

    # ── Final recommendation ──
    if score >= 5:
        recommendation = "STRONG BUY"
        action_color   = "buy_strong"
    elif score >= 2:
        recommendation = "BUY"
        action_color   = "buy"
    elif score <= -5:
        recommendation = "STRONG SELL"
        action_color   = "sell_strong"
    elif score <= -2:
        recommendation = "SELL / AVOID"
        action_color   = "sell"
    else:
        recommendation = "HOLD / WATCH"
        action_color   = "hold"

    # ── Buy/sell timing guidance ──
    buy_when  = []
    sell_when = []

    if rsi > 30:
        buy_when.append(f"RSI drops below 30 (currently {rsi:.0f})")
    if ma20:
        buy_when.append(f"Price pulls back to MA20 support (~${ma20:.2f})")
    buy_when.append("MACD bullish crossover confirmed on daily chart")

    if rsi > 60:
        sell_when.append(f"RSI exceeds 75 (currently {rsi:.0f}) — take partial profit")
    if ma50:
        sell_when.append(f"Price hits MA50 resistance (~${ma50:.2f}) if in downtrend")
    sell_when.append("MACD bearish crossover or price breaks below MA20 on volume")

    return {
        "ticker":         ticker,
        "name":           name,
        "price":          close,
        "score":          score,
        "recommendation": recommendation,
        "action_color":   action_color,
        "signals":        signals,
        "cautions":       cautions,
        "buy_when":       buy_when,
        "sell_when":      sell_when,
        "rsi":            rsi,
        "week_chg":       week_chg,
        "month_chg":      month_chg,
        "ma20":           ma20,
        "ma50":           ma50,
        "ma200":          ma200,
        "volume_spike":   bool(row["Vol_Spike"]),
    }

def fetch_stock(ticker, name, price_max):
    try:
        tk   = yf.Ticker(ticker)
        df   = tk.history(period="1y")
        info = tk.info

        if df.empty:
            print(f"  ⚠  No data for {ticker}")
            return None

        close = float(df["Close"].iloc[-1])

        if close > price_max:
            print(f"  ⏭  {ticker} ${close:.2f} exceeds max ${price_max:.2f} — included anyway (watchlist override)")

        df = calc_indicators(df)
        sig = generate_signals(df, ticker, name)

        # Extra info from yfinance
        sig["sector"]    = info.get("sector", "N/A")
        sig["industry"]  = info.get("industry", "N/A")
        sig["mkt_cap"]   = info.get("marketCap", 0)
        sig["pe_ratio"]  = info.get("trailingPE", None)
        sig["div_yield"] = info.get("dividendYield", None)
        sig["52w_high"]  = info.get("fiftyTwoWeekHigh", None)
        sig["52w_low"]   = info.get("fiftyTwoWeekLow", None)
        sig["news"]      = tk.news[:3] if hasattr(tk, "news") else []

        # Next dividend
        try:
            cal = tk.calendar
            sig["next_div_date"] = str(cal.get("Dividend Date", "N/A")) if cal else "N/A"
        except Exception:
            sig["next_div_date"] = "N/A"

        print(f"  ✓  {ticker:10s}  ${close:.2f}  [{sig['recommendation']}]  score={sig['score']}")
        return sig

    except Exception as e:
        print(f"  ✗  {ticker} error: {e}")
        return None

# GLOBAL INDEX CORRELATION
# ─────────────────────────────────────────────
def fetch_index_correlation():
    print("\n📊 Fetching global indices...")
    data = {}
    for name, symbol in GLOBAL_INDICES.items():
        try:
            df = yf.download(symbol, period="3mo", progress=False)
            if not df.empty:
                close = df["Close"].squeeze()
                ret   = close.pct_change().dropna()
                data[name] = {
                    "symbol":    symbol,
                    "last":      float(close.iloc[-1]),
                    "chg_1d":    float((close.iloc[-1] - close.iloc[-2]) / close.iloc[-2] * 100),
                    "chg_1mo":   float((close.iloc[-1] - close.iloc[-22]) / close.iloc[-22] * 100) if len(close) >= 22 else 0,
                    "returns":   ret,
                }
        except Exception as e:
            print(f"  ⚠ {name}: {e}")

    # Compute correlation matrix between ASX 200 and others
    corr_summary = []
    if "ASX 200" in data:
        asx_ret = data["ASX 200"]["returns"]
        for idx_name, idx_data in data.items():
            if idx_name == "ASX 200":
                continue
            try:
                aligned = pd.concat([asx_ret, idx_data["returns"]], axis=1).dropna()
                if len(aligned) > 10:
                    corr = aligned.iloc[:, 0].corr(aligned.iloc[:, 1])
                    corr_summary.append({
                        "index": idx_name,
                        "corr":  round(corr, 3),
                        "chg_1d":  idx_data["chg_1d"],
                        "chg_1mo": idx_data["chg_1mo"],
                        "last":    idx_data["last"],
                    })
            except Exception:
                pass

    return data, corr_summary

class TradingReport(FPDF):
    def header(self):
        self.set_fill_color(15, 23, 42)
        self.rect(0, 0, 210, 18, "F")
        self.set_font("Helvetica", "B", 11)
        self.set_text_color(255, 255, 255)
        self.cell(0, 18, "  TRADING ANALYSER — NIGHTLY REPORT", ln=True)
        self.set_text_color(0, 0, 0)

    def footer(self):
        self.set_y(-12)
        self.set_font("Helvetica", "I", 7)
        self.set_text_color(120, 120, 120)
        self.cell(0, 10,
                  "⚠  This report is for informational purposes only and does not constitute financial advice. "
                  "Always do your own research before investing.  |  "
                  f"Page {self.page_no()}",
                  align="C")


def rec_colors(action_color):
    colors = {
        "buy_strong":  (22, 163, 74),
        "buy":         (74, 222, 128),
        "hold":        (234, 179, 8),
        "sell":        (249, 115, 22),
        "sell_strong": (220, 38, 38),
    }
    return colors.get(action_color, (100, 100, 100))

        pdf.ln(3)
        pdf.set_font("Helvetica", "B", 10)
        pdf.set_text_color(15, 23, 42)

        pdf.set_draw_color(*c)
        pdf.line(10, pdf.get_y(), 200, pdf.get_y())
        pdf.ln(2)

        col_w = 90
        pdf.set_font("Helvetica", "B", 8)
        pdf.set_fill_color(22, 163, 74)
        pdf.set_text_color(255, 255, 255)
        pdf.cell(col_w, 6, "  BUY WHEN:", fill=True)
        pdf.set_fill_color(220, 38, 38)
        pdf.cell(col_w, 6, "  SELL / TAKE PROFIT WHEN:", fill=True)
        pdf.ln()

        pdf.set_font("Helvetica", "", 8)
        pdf.set_text_color(30, 30, 30)
        max_rows = max(len(r["buy_when"]), len(r["sell_when"]))
        for i in range(max_rows):
            buy_txt  = f"  • {r['buy_when'][i]}"  if i < len(r["buy_when"])  else ""
            sell_txt = f"  • {r['sell_when'][i]}" if i < len(r["sell_when"]) else ""

            pdf.set_fill_color(240, 253, 244)
            pdf.multi_cell(col_w, 5, buy_txt, fill=True)

            pdf.set_xy(10 + col_w, y_before)
            pdf.set_fill_color(255, 241, 242)
            pdf.multi_cell(col_w, 5, sell_txt, fill=True)
            pdf.set_y(max(y_after, pdf.get_y()))

            pdf.ln(3)
            pdf.set_font("Helvetica", "B", 8)
            pdf.set_text_color(15, 23, 42)

                f"Dividend Yield: {r['div_yield']*100:.2f}%   |   Next Dividend Date: {r.get('next_div_date','N/A')}",
                ln=True)

            pdf.set_font("Helvetica", "B", 9)
            pdf.set_text_color(15, 23, 42)
            pdf.cell(0, 6, "Recent News", ln=True)
            pdf.set_font("Helvetica", "", 8)
            pdf.set_text_color(50, 50, 50)
            for article in r["news"][:3]:
                title = article.get("content", {}).get("title", article.get("title", ""))
                if title:
                    pub = article.get("content", {}).get("pubDate", "")
                    pdf.multi_cell(0, 5, f"• {title[:120]}  [{pub[:10]}]")

    # ── Disclaimer page ─────────────────────────────────────────────
    pdf.add_page()
    pdf.set_font("Helvetica", "B", 12)
    pdf.set_text_color(15, 23, 42)

# SAVE JSON FOR DASHBOARD
# ─────────────────────────────────────────────
def save_dashboard_json(results, corr_summary):
    safe_results = []
    for r in results:
        if not r:
            continue
        safe = {k: v for k, v in r.items() if k != "news"}
        safe["news"] = [
            {"title": a.get("content", {}).get("title", a.get("title", "")),
             "date":  a.get("content", {}).get("pubDate", "")[:10]}
            for a in r.get("news", [])[:3]
        ]
        safe_results.append(safe)

    payload = {
        "generated":     datetime.now().isoformat(),
        "results":       safe_results,
        "correlations":  corr_summary,
    }
    out = Path(__file__).parent.parent / "reports" / "latest.json"
    with open(out, "w") as f:
        json.dump(payload, f, indent=2, default=str)
    print(f"✅ Dashboard JSON saved → {out}")
